1. help

// shows a list of possible user commands and their respective descriptions
help

// show a list of detailed instruction within a command word
help <command word>
e.g help add 



2. add

// add a floating task
add <task name>

// add a task with a deadline
add <task name> by~ <date>

// add a task with timed duration
add <task name> from~ <start date> to~ <due date>




3. delete

// delete a task
delete <task name>

// delete a task
delete <task index>

// delete start date from a task
delete start date from~ <task index>

// delete due date from a task
delete due date from~ <task index>




4. change 

// rename existing task
change <task index> to~ <new task name>

// change start date for a existing task
change start date to~ <new date> from~ <task index>

// change due date for a existing task
change due date to~ <new date> from~ <task index>



5. search

// search for tasks containing certain keywords
search <keywords>

// search for tasks within a specific range of dates
search from~ <start date> to~ <due date>



6. complete

// mark a task as completed
complete <task name>

// mark a task as completed
complete <task index>

7.

// go to a particular section
goto <tagname>
